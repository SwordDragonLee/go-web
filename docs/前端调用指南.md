# 前端调用登录接口指南

## 接口信息

### 登录接口

- **URL**: `POST /api/v1/auth/login`
- **完整地址**: `http://localhost:8080/api/v1/auth/login`
- **Content-Type**: `application/json`
- **认证**: 无需认证（公开接口）

### 请求参数

```json
{
  "username": "testuser",  // 用户名/邮箱/手机号
  "password": "123456"      // 密码
}
```

### 响应格式

**成功响应（200）**:
```json
{
  "code": 200,
  "message": "success",
  "data": {
    "token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...",
    "token_type": "Bearer",
    "expires_in": 604800,
    "user": {
      "id": 1,
      "username": "testuser",
      "email": "test@example.com",
      "phone": "13800138000",
      "nickname": "测试用户",
      "avatar": "",
      "status": 1,
      "role": "user",
      "last_login_at": "2024-01-01T12:00:00Z"
    }
  }
}
```

**失败响应（400）**:
```json
{
  "code": 400,
  "message": "用户不存在",
  "data": null
}
```

## 前端调用方式

### 1. 使用 Fetch API（原生 JavaScript）

```javascript
// 登录函数
async function login(username, password) {
  try {
    const response = await fetch('http://localhost:8080/api/v1/auth/login', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        username: username,
        password: password
      })
    });

    const result = await response.json();

    if (result.code === 200) {
      // 登录成功
      const { token, token_type, user } = result.data;
      
      // 保存 Token 到 localStorage
      localStorage.setItem('token', token);
      localStorage.setItem('token_type', token_type);
      localStorage.setItem('user', JSON.stringify(user));
      
      console.log('登录成功', user);
      return result.data;
    } else {
      // 登录失败
      console.error('登录失败:', result.message);
      throw new Error(result.message);
    }
  } catch (error) {
    console.error('请求错误:', error);
    throw error;
  }
}

// 使用示例
login('testuser', '123456')
  .then(data => {
    console.log('登录成功', data);
  })
  .catch(error => {
    console.error('登录失败', error);
  });
```

### 2. 使用 Axios（推荐）

#### 安装 Axios

```bash
npm install axios
# 或
yarn add axios
```

#### 基础调用

```javascript
import axios from 'axios';

// 登录函数
async function login(username, password) {
  try {
    const response = await axios.post('http://localhost:8080/api/v1/auth/login', {
      username: username,
      password: password
    });

    if (response.data.code === 200) {
      const { token, token_type, user } = response.data.data;
      
      // 保存 Token
      localStorage.setItem('token', token);
      localStorage.setItem('user', JSON.stringify(user));
      
      // 设置 Axios 默认请求头
      axios.defaults.headers.common['Authorization'] = `${token_type} ${token}`;
      
      return response.data.data;
    } else {
      throw new Error(response.data.message);
    }
  } catch (error) {
    if (error.response) {
      // 服务器返回了错误响应
      console.error('登录失败:', error.response.data.message);
    } else {
      // 请求失败（网络错误等）
      console.error('请求错误:', error.message);
    }
    throw error;
  }
}

// 使用示例
login('testuser', '123456');
```

#### 配置 Axios 实例（推荐）

```javascript
import axios from 'axios';

// 创建 Axios 实例
const api = axios.create({
  baseURL: 'http://localhost:8080/api/v1',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
});

// 请求拦截器 - 自动添加 Token
api.interceptors.request.use(
  config => {
    const token = localStorage.getItem('token');
    const tokenType = localStorage.getItem('token_type') || 'Bearer';
    
    if (token) {
      config.headers.Authorization = `${tokenType} ${token}`;
    }
    
    return config;
  },
  error => {
    return Promise.reject(error);
  }
);

// 响应拦截器 - 统一处理错误
api.interceptors.response.use(
  response => {
    // 如果响应成功，直接返回 data
    if (response.data.code === 200) {
      return response.data.data;
    } else {
      // 业务错误
      return Promise.reject(new Error(response.data.message));
    }
  },
  error => {
    // HTTP 错误（401, 403, 500 等）
    if (error.response?.status === 401) {
      // Token 过期或无效，跳转到登录页
      localStorage.removeItem('token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

// 登录函数
export async function login(username, password) {
  const data = await api.post('/auth/login', {
    username,
    password
  });
  
  // 保存 Token
  localStorage.setItem('token', data.token);
  localStorage.setItem('token_type', data.token_type);
  localStorage.setItem('user', JSON.stringify(data.user));
  
  // 更新 Axios 默认请求头
  api.defaults.headers.common['Authorization'] = `${data.token_type} ${data.token}`;
  
  return data;
}

// 使用示例
login('testuser', '123456')
  .then(data => {
    console.log('登录成功', data.user);
  })
  .catch(error => {
    console.error('登录失败', error.message);
  });
```

### 3. React 示例

```jsx
import React, { useState } from 'react';
import axios from 'axios';

function LoginForm() {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');

  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);
    setError('');

    try {
      const response = await axios.post('http://localhost:8080/api/v1/auth/login', {
        username,
        password
      });

      if (response.data.code === 200) {
        const { token, token_type, user } = response.data.data;
        
        // 保存 Token
        localStorage.setItem('token', token);
        localStorage.setItem('user', JSON.stringify(user));
        
        // 设置默认请求头
        axios.defaults.headers.common['Authorization'] = `${token_type} ${token}`;
        
        // 跳转到首页
        window.location.href = '/';
      } else {
        setError(response.data.message);
      }
    } catch (err) {
      setError(err.response?.data?.message || '登录失败，请重试');
    } finally {
      setLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>用户名/邮箱/手机号:</label>
        <input
          type="text"
          value={username}
          onChange={(e) => setUsername(e.target.value)}
          required
        />
      </div>
      <div>
        <label>密码:</label>
        <input
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          required
        />
      </div>
      {error && <div style={{ color: 'red' }}>{error}</div>}
      <button type="submit" disabled={loading}>
        {loading ? '登录中...' : '登录'}
      </button>
    </form>
  );
}

export default LoginForm;
```

### 5. jQuery 示例

```javascript
function login(username, password) {
  $.ajax({
    url: 'http://localhost:8080/api/v1/auth/login',
    method: 'POST',
    contentType: 'application/json',
    data: JSON.stringify({

      }
    },
    error: function(xhr, status, error) {
      console.error('请求错误:', error);
    }
  });
}
```

```javascript
// Fetch API
fetch('http://localhost:8080/api/v1/user/profile', {
  method: 'GET',

axios.get('http://localhost:8080/api/v1/user/profile', {

  }

  .then(response => console.log(response.data));
```

### 方式 2：设置默认请求头（推荐）

```javascript
// Axios 全局配置
axios.defaults.headers.common['Authorization'] = `Bearer ${localStorage.getItem('token')}`;

// 之后所有请求都会自动包含 Token
axios.get('http://localhost:8080/api/v1/user/profile')
  .then(response => console.log(response.data));
```

### 方式 3：使用 Axios 拦截器（最佳实践）

```javascript
// 请求拦截器 - 自动添加 Token
axios.interceptors.request.use(
  config => {
    const token = localStorage.getItem('token');
    if (token) {

    return Promise.reject(error);

// 响应拦截器 - 处理 Token 过期

  }
);
```

## 完整示例：登录 + 获取用户信息

```javascript
// api.js - API 封装
import axios from 'axios';

const api = axios.create({
  baseURL: 'http://localhost:8080/api/v1',
  timeout: 10000
});

// 请求拦截器
api.interceptors.request.use(
  config => {
    const token = localStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  error => Promise.reject(error)
);

// 响应拦截器
api.interceptors.response.use(
  response => {
    if (response.data.code === 200) {
      return response.data.data;
    } else {
      return Promise.reject(new Error(response.data.message));
    }
  },
  error => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

// API 方法
export const authAPI = {
  // 登录
  login: (username, password) => {
    return api.post('/auth/login', { username, password }).then(data => {
      localStorage.setItem('token', data.token);
      localStorage.setItem('user', JSON.stringify(data.user));
      return data;
    });
  },
  
  // 注册
  register: (userData) => {
    return api.post('/auth/register', userData);
  },
  
  // 获取个人信息（需要 Token）
  getProfile: () => {
    return api.get('/user/profile');
  },
  
  // 更新个人信息（需要 Token）
  updateProfile: (data) => {
    return api.put('/user/profile', data);
  }
};

// 使用示例
async function handleLogin() {
  try {
    const data = await authAPI.login('testuser', '123456');
    console.log('登录成功', data.user);
    
    // 登录后获取用户信息
    const profile = await authAPI.getProfile();
    console.log('用户信息', profile);
  } catch (error) {
    console.error('登录失败', error.message);
  }
}
```

## 常见问题

### 1. CORS 跨域问题

如果前端和后端不在同一域名，需要配置 CORS：

```go
// 在 router.go 中添加 CORS 中间件
import "github.com/gin-contrib/cors"

r.Use(cors.New(cors.Config{
    AllowOrigins:     []string{"http://localhost:3000"}, // 前端地址
    AllowMethods:     []string{"GET", "POST", "PUT", "DELETE", "OPTIONS"},
    AllowHeaders:     []string{"Origin", "Content-Type", "Authorization"},
    ExposeHeaders:    []string{"Content-Length"},
    AllowCredentials: true,
}))
```

### 2. Token 过期处理

```javascript
// 检查 Token 是否过期
function isTokenExpired() {
  const token = localStorage.getItem('token');
  if (!token) return true;
  
  try {
    const payload = JSON.parse(atob(token.split('.')[1]));
    const exp = payload.exp * 1000; // 转换为毫秒
    return Date.now() >= exp;
  } catch {
    return true;
  }
}

// 刷新 Token
async function refreshToken() {
  try {
    const response = await axios.post('/api/v1/user/refresh-token');
    localStorage.setItem('token', response.data.data.token);
    return response.data.data.token;
  } catch (error) {
    // 刷新失败，跳转登录
    localStorage.removeItem('token');
    window.location.href = '/login';
  }
}
```

### 3. 错误处理

```javascript
// 统一错误处理
function handleApiError(error) {
  if (error.response) {
    switch (error.response.status) {
      case 400:
        return '请求参数错误';
      case 401:
        return '未授权，请重新登录';
      case 403:
        return '权限不足';
      case 404:
        return '资源不存在';
      case 500:
        return '服务器错误';
      default:
        return error.response.data?.message || '请求失败';
    }
  } else {
    return '网络错误，请检查网络连接';
  }
}
```

## 总结

### 登录接口调用步骤

1. **发送登录请求**
   ```javascript
   POST /api/v1/auth/login
   Body: { username, password }
   ```

2. **保存 Token**
   ```javascript
   localStorage.setItem('token', token);
   ```

3. **后续请求携带 Token**
   ```javascript
   Headers: { Authorization: 'Bearer ' + token }
   ```

### 最佳实践

- ✅ 使用 Axios 拦截器自动添加 Token
- ✅ 统一处理错误响应
- ✅ Token 过期自动跳转登录
- ✅ 使用环境变量管理 API 地址
- ✅ 封装 API 方法，便于复用
